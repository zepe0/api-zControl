-- cliente_id

(
  SELECT IF(
    EXISTS(
      SELECT 1
      FROM INFORMATION_SCHEMA.COLUMNS
      WHERE TABLE_SCHEMA = @db
        AND TABLE_NAME = 'Albaranes'
        AND COLUMN_NAME = 'cliente_id'
    ),
    'SELECT "cliente_id ya existe" AS info;',
    'ALTER TABLE `Albaranes` ADD COLUMN `cliente_id` VARCHAR(255) NULL AFTER `fecha`;' 
  )
);

DEALLOCATE PREPARE stmt;
